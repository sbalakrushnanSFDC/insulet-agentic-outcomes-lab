### Automated SOQL Checks (executable against orgsyncng post-deploy)

```sql
-- SQ-01: Inside Sales ServicePresenceStatus records exist (3 expected)
SELECT Id, DeveloperName, MasterLabel
FROM ServicePresenceStatus
WHERE DeveloperName IN ('IS_Available_Voice', 'IS_Busy_Voice', 'IS_ACW')

-- SQ-02: IS_CTI_Agent_Access PSG contains 3 permission sets
SELECT PermissionSet.Name
FROM PermissionSetGroupComponent
WHERE PermissionSetGroup.DeveloperName = 'IS_CTI_Agent_Access'

-- SQ-03: IS_CTI_Base permission set exists with correct object permissions
SELECT SobjectType, PermissionsRead, PermissionsCreate, PermissionsEdit, PermissionsDelete
FROM ObjectPermissions
WHERE Parent.Name = 'IS_CTI_Base'

-- SQ-04: Inside_Sales_Voice queue exists
SELECT Id, DeveloperName, Name
FROM Group
WHERE Type = 'Queue' AND DeveloperName = 'Inside_Sales_Voice'

-- SQ-05: HCP_Inbound_Voice queue exists
SELECT Id, DeveloperName, Name
FROM Group
WHERE Type = 'Queue' AND DeveloperName = 'HCP_Inbound_Voice'

-- SQ-06: IS_Softphone_Layout exists
SELECT Id, Name
FROM SoftphoneLayout
WHERE Name = 'IS_Softphone_Layout'

-- SQ-07: Task.Call_Disposition__c field deployed (Phase 2)
SELECT Id, QualifiedApiName, DataType
FROM FieldDefinition
WHERE EntityDefinition.QualifiedApiName = 'Task'
  AND QualifiedApiName = 'Call_Disposition__c'

-- SQ-08: VoiceCall.Recording_Opted_Out__c field deployed (Phase 2)
SELECT Id, QualifiedApiName, DataType
FROM FieldDefinition
WHERE EntityDefinition.QualifiedApiName = 'VoiceCall'
  AND QualifiedApiName = 'Recording_Opted_Out__c'

-- SQ-09: VoiceCall records post-test (run after live call test)
SELECT Id, CallType, CallStartDateTime, CallEndDateTime, CallDurationInSeconds,
       Call_Category__c, True_Connection__c, Recording_Opted_Out__c
FROM VoiceCall
WHERE CreatedDate = TODAY
ORDER BY CallStartDateTime DESC
LIMIT 10

-- SQ-10: Task records created by VoiceCallActivityHandler post-test
SELECT Id, Subject, WhoId, WhatId, Call_Disposition__c, Call_Duration_Seconds__c,
       CallType, CreatedDate
FROM Task
WHERE Subject LIKE '%Call%'
  AND CreatedDate = TODAY
ORDER BY CreatedDate DESC
LIMIT 10

-- SQ-11: PSG assignment check for Inside Sales test user
SELECT Assignee.Name, PermissionSetGroup.DeveloperName
FROM PermissionSetAssignment
WHERE PermissionSetGroup.DeveloperName = 'IS_CTI_Agent_Access'
LIMIT 10

-- SQ-12: ServiceChannel verification (shared with NGPSTE-132)
SELECT Id, DeveloperName, RelatedEntity
FROM ServiceChannel
WHERE DeveloperName = 'sfdc_phone'
```
